def group(priority) #returns array of each group
  group = Array.new
  case priority
  when "high"
    priority = 1
  when "medium"
    priority = 2
  when "low"
    priority = 3
  end
  @@list.each do |ai|
    if ai.priority == priority
      group << ai
    end
  end
  return group
end

def list
  #convert list of strings into AI, and put it in an array
  #skip if already created
  if @@list == nil
    file = File.open("list_of_available_ingredients.txt")
    @@list = Array.new
    file.each_line do |line|
      temp = Array.new
      temp = line.chomp.split(/\s\s/) #fails when the file isn't formatted properly
      temp[1] = Date.strptime(temp[1], "%Y/%m/%d")
      @@list << self.new(temp[0], temp[1])
    end
    file.close
    #sort the list by days left
    @@list.sort_by! do |i|
      i.days_left
    end
  end
  return @@list
end

def initialize(name, date_of_purchase)
  @name = name
  @date_of_purchase = date_of_purchase
  @expiration = self.get_expiration
  @days_left = (@expiration - Date.today).to_i
  case @days_left
  when (0..1)
    @priority = 1
  when (2..3)
    @priority = 2
  when (4..7)
    @priority = 3
  else
    @priority = 4
  end
end

def get_expiration
  i = 0
  KnownIngredient.list.each do |ki|
    if ki.name == @name
      return @date_of_purchase + ki.life
      i += 1
      break
    end
  end
  if i == 0
    puts "unknown ingredient: #{@name}"
    puts "How many days does it last after purchase? : "
    return @date_of_purchase + gets.chomp.to_i
  end
end
